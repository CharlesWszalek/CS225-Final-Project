import sympy as sym
from sympy.matrices import Matrix

sym.init_printing()
hp, hx, a = sym.symbols('h+, hx, a')
# M = Matrix(4, 4, [0, 0, 0, 0, 0, hp, hx, 0, 0, hx, -hp, 0, 0, 0, 0, 0])
M = Matrix([[-1, 0,        0,         0],
            [0,  a*hp + 1, a*hx,      0],
            [0,  a*hx,     -a*hp + 1, 0],
            [0,  0,        0,         1]])
M_i = sym.simplify(M.inv())
sym.pprint(M_i)
